 0390        /// Calculates contrast ratio between two colors (WCAG 2.1 formula)
 0391        /// </summary>
 0392        /// <param name="color1">First color</param>
 0393        /// <param name="color2">Second color</param>
 0394        /// <returns>Contrast ratio (1:1 to 21:1)</returns>
 0395        public static double CalculateContrastRatio(Color color1, Color color2)
 0396        {
 0397            double luminance1 = CalculateRelativeLuminance(color1);
 398            double luminance2 = CalculateRelativeLuminance(color2);
 399
 400            // Ensure the lighter color is always in the numerator
 401            if (luminance1 <= luminance2)
 402            {
 403                (luminance1, luminance2) = (luminance2, luminance1);
 404            }

 406            return (luminance1 + 0.05) / (luminance2 + 0.05);

 409        /// <summary>
 410        /// Calculates relative luminance of a color (WCAG 2.1 formula)
 411        /// </summary>
 412        /// <param name="color">Color to calculate</param>
 413        /// <returns>Relative luminance (0 to 1)</returns>
 414        private static double CalculateRelativeLuminance(Color color)

 432        /// <summary>
 433        /// Ensures WCAG 2.1 AA compliant contrast by adjusting colors if needed
 434        /// </summary>
 435        /// <param name="textColor">Text color</param>
 436        /// <param name="backgroundColor">Background color</param>
 437        /// <param name="minRatio">Minimum required contrast ratio (4.5:1 for normal text, 3:1 for large text)</param>
 438        /// <returns>Adjusted text color to meet contrast requirements</returns>
 439        public static Color EnsureAccessibleContrast(Color textColor, Color backgroundColor, double minRatio = 4.5)
 440        {
 441            // Calculate current contrast ratio
 442            double ratio = CalculateContrastRatio(textColor, backgroundColor);
 443
 444            // If already compliant, return original color
 445            if (ratio >= minRatio)
 446            {
 447                return textColor;
 448            }
 449
 450            // Determine if we're dealing with light or dark background
 451            bool isDarkBackground = CalculateRelativeLuminance(backgroundColor) < 0.5;
 452
 453            // Adjust the text color to meet contrast requirements
 454            Color adjustedColor = textColor;
 455            int step = isDarkBackground ? 5 : -5; // Lighten for dark backgrounds, darken for light
 456
 457            while (ratio < minRatio)
 458            {
 459                int r = Math.Clamp(adjustedColor.R + step, 0, 255);
 460                int g = Math.Clamp(adjustedColor.G + step, 0, 255);
 461                int b = Math.Clamp(adjustedColor.B + step, 0, 255);
 462
 463                adjustedColor = Color.FromArgb(adjustedColor.A, r, g, b);
 464                ratio = CalculateContrastRatio(adjustedColor, backgroundColor);
 465
 466                // Break if we've gone as far as possible but still can't meet requirements
 467                if ((isDarkBackground && r == 255 && g == 255 && b == 255) ||
 468                    (!isDarkBackground && r == 0 && g == 0 && b == 0))
 469                {
 470                    break;
 471                }
 472            }
 473
 474            return adjustedColor;
 475        }
